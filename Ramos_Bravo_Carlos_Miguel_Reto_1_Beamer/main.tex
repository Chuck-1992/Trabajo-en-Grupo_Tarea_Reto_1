\section{Contexto y objetivo}
% ---------------------------------------------------------

\begin{frame}{Objetivo de la actividad}
\begin{itemize}
    \item Calcular determinantes de orden $2\times2$, $3\times3$ y $4\times4$.
    \item Comprender el uso de menores y cofactores.
    \item Aplicar la Regla de Cramer para resolver un sistema lineal.
    \item Interpretar la solución en un caso aplicado a ingeniería y ciberseguridad.
\end{itemize}

\begin{block}{Idea central}
Si el determinante principal cumple $D\neq0$, el sistema tiene solución única y puede resolverse mediante la Regla de Cramer.
\end{block}
\end{frame}

\begin{frame}{Problema aplicado: control de acceso}
\justifying
En un laboratorio de computación se registran eventos de acceso. El sistema no entrega directamente el número de cada evento, sino reportes combinados. Por eso se plantea un sistema lineal con cuatro incógnitas.

\vspace{0.25cm}
\begin{table}
\centering
\begin{tabular}{cclc}
\toprule
\# & Variable & Significado & Unidad \\
\midrule
1 & $x$ & Accesos correctos & eventos \\
2 & $y$ & Contraseñas incorrectas & eventos \\
3 & $z$ & Bloqueos del sistema & eventos \\
4 & $w$ & Accesos remotos & eventos \\
\bottomrule
\end{tabular}
\end{table}

\begin{alertblock}{Pregunta guía}
¿Cuántos eventos de cada tipo ocurrieron durante la jornada analizada?
\end{alertblock}
\end{frame}

\begin{frame}{Recopilación de datos del sistema}
\begin{table}
\centering
\scriptsize
\begin{tabular}{clp{8.7cm}}
\toprule
\# & Reporte & Interpretación \\
\midrule
1 & $x+y+z+w=95$ & Total general de eventos registrados. \\
2 & $2x+y+w=125$ & Reporte ponderado donde los accesos correctos tienen doble peso. \\
3 & $x+2y+z=110$ & Reporte ponderado donde las contraseñas incorrectas tienen doble peso. \\
4 & $y+2z+3w=85$ & Reporte ponderado donde los bloqueos y accesos remotos tienen mayor peso. \\
\bottomrule
\end{tabular}
\end{table}

\begin{block}{Utilidad del modelo}
Los reportes combinados se transforman en ecuaciones. Así se puede encontrar la cantidad real de cada evento.
\end{block}
\end{frame}

\begin{frame}{Sistema de ecuaciones lineales}
\[
\begin{cases}
x+y+z+w=95 \\
2x+y+w=125 \\
x+2y+z=110 \\
y+2z+3w=85
\end{cases}
\]

\begin{block}{Lectura del sistema}
Cada ecuación representa una forma distinta de contar o ponderar los eventos del laboratorio.
\end{block}
\end{frame}

\begin{frame}{Forma matricial}
El sistema se expresa como $A\cdot X=B$:

\[
A=\mat{
1&1&1&1\\
2&1&0&1\\
1&2&1&0\\
0&1&2&3}
\qquad
X=\mat{x\\y\\z\\w}
\qquad
B=\mat{95\\125\\110\\85}
\]

\begin{alertblock}{Condición para Cramer}
Antes de resolver, se calcula $D=\det(A)$. Si $D\neq0$, existe una única solución.
\end{alertblock}
\end{frame}

% ---------------------------------------------------------
\section{Determinantes, menores y cofactores}
% ---------------------------------------------------------

\begin{frame}{Determinante de orden $2\times2$}
Para una matriz de orden $2\times2$:
\[
\detm{a&b\\c&d}=ad-bc
\]

\begin{exampleblock}{Ejemplo}
\[
\detm{3&1\\2&4}=3(4)-1(2)=12-2=10
\]
\end{exampleblock}

\begin{block}{Importancia}
Los determinantes $2\times2$ aparecen dentro del cálculo de determinantes $3\times3$ y $4\times4$.
\end{block}
\end{frame}

\begin{frame}{Determinante de orden $3\times3$}
Para una matriz $3\times3$ se puede expandir por una fila o columna:
\[
\detm{a&b&c\\d&e&f\\g&h&i}
=a\detm{e&f\\h&i}-b\detm{d&f\\g&i}+c\detm{d&e\\g&h}
\]

\begin{exampleblock}{Ejemplo breve}
\[
\detm{1&0&1\\2&1&0\\1&2&3}
=1\detm{1&0\\2&3}+1\detm{2&1\\1&2}=3+3=6
\]
\end{exampleblock}
\end{frame}

\begin{frame}{Menores y cofactores}
\begin{block}{Menor $M_{ij}$}
Es el determinante que queda al eliminar la fila $i$ y la columna $j$ de un elemento de la matriz.
\end{block}

\begin{block}{Cofactor $C_{ij}$}
El cofactor es el menor con el signo correspondiente:
\[
C_{ij}=(-1)^{i+j}M_{ij}
\]
\end{block}

\begin{exampleblock}{Patrón de signos}
\[
\mat{+&-&+&-\\-&+&-&+\\+&-&+&-\\-&+&-&+}
\]
\end{exampleblock}
\end{frame}

\begin{frame}[shrink=5]{Ejemplo de menor y cofactor en la tercera fila}
Tomando como referencia el elemento $a_{31}=1$ de la matriz $A$:

\[
A=\mat{
1&1&1&1\\
2&1&0&1\\
\color{azulPUCE}{1}&2&1&0\\
0&1&2&3}
\]

Se elimina la fila 3 y la columna 1:
\[
M_{31}=\detm{1&1&1\\1&0&1\\1&2&3}=-2
\]

Como la posición es $(3,1)$:
\[
C_{31}=(-1)^{3+1}M_{31}=1(-2)=\boxed{-2}
\]
\end{frame}

\begin{frame}[shrink=8]{Determinante principal $D$}
\[
D=\det(A)=
\detm{
1&1&1&1\\
2&1&0&1\\
1&2&1&0\\
0&1&2&3}
\]

Expansión por la primera fila:
\[
D=1M_{11}-1M_{12}+1M_{13}-1M_{14}
\]

\begin{block}{Menores necesarios}
\[
M_{11}=6,\qquad M_{12}=8,\qquad M_{13}=10,\qquad M_{14}=4
\]
\end{block}

\[
D=6-8+10-4=\boxed{4}
\]
\end{frame}

\begin{frame}[shrink=6]{Cálculo del menor $M_{11}$}
Se elimina la fila 1 y la columna 1:
\[
M_{11}=\detm{1&0&1\\2&1&0\\1&2&3}
\]

\[
M_{11}=1\detm{1&0\\2&3}-0\detm{2&0\\1&3}+1\detm{2&1\\1&2}
\]

\[
M_{11}=1[(1)(3)-(0)(2)]+1[(2)(2)-(1)(1)]
\]

\[
M_{11}=3+3=\boxed{6}
\]
\end{frame}

\begin{frame}[shrink=8]{Cálculo de $M_{12}$, $M_{13}$ y $M_{14}$}
\[
M_{12}=\detm{2&0&1\\1&1&0\\0&2&3}=\boxed{8}
\]

\[
M_{13}=\detm{2&1&1\\1&2&0\\0&1&3}=\boxed{10}
\]

\[
M_{14}=\detm{2&1&0\\1&2&1\\0&1&2}=\boxed{4}
\]

\begin{alertblock}{Resultado importante}
Como $D=4\neq0$, el sistema tiene solución única y se puede aplicar la Regla de Cramer.
\end{alertblock}
\end{frame}

\section{Regla de Cramer}
% ---------------------------------------------------------

\begin{frame}{Proceso de resolución con Cramer}
\begin{center}
\begin{tikzpicture}[
node distance=0.9cm and 1.05cm,
box/.style={rectangle,rounded corners,draw=azulPUCE,very thick,fill=celesteSuave,align=center,minimum width=2.2cm,minimum height=0.75cm,font=\small},
arrow/.style={-{Latex[length=3mm]},very thick,azulPUCE}
]
\node[box] (a) {Sistema\\lineal};
\node[box,right=of a] (b) {Matriz\\$A$ y $B$};
\node[box,right=of b] (c) {Calcular\\$D$};
\node[box,below=of b] (d) {Calcular\\$D_x,D_y,D_z,D_w$};
\node[box,right=of d] (e) {Dividir\\por $D$};
\draw[arrow] (a) -- (b);
\draw[arrow] (b) -- (c);
\draw[arrow] (c) |- (d);
\draw[arrow] (d) -- (e);
\end{tikzpicture}
\end{center}

\[
x=\frac{D_x}{D},\qquad y=\frac{D_y}{D},\qquad z=\frac{D_z}{D},\qquad w=\frac{D_w}{D}
\]
\end{frame}

\begin{frame}[shrink=6]{Cálculo de $D_x$}
Se reemplaza la columna de $x$ por $B$:
\[
D_x=\detm{
95&1&1&1\\
125&1&0&1\\
110&2&1&0\\
85&1&2&3}
\]

\[
D_x=95(6)-1(510)+1(360)-1(240)
\]

\[
D_x=570-510+360-240=\boxed{180}
\]

\[
x=\frac{D_x}{D}=\frac{180}{4}=\boxed{45}
\]
\end{frame}

% Desarrollo completo agregado para explicar de dónde sale el valor de Dx.
\begin{frame}[shrink=10]{Desarrollo completo de $D_x$: expansión por cofactores}
Se reemplaza la columna de $x$ por el vector $B$:
\[
D_x=\detm{
95&1&1&1\\
125&1&0&1\\
110&2&1&0\\
85&1&2&3}
\]

Se expande por la primera columna:
\[
\begin{aligned}
D_x={}&95\detm{1&0&1\\2&1&0\\1&2&3}
-125\detm{1&1&1\\2&1&0\\1&2&3}\\
&+110\detm{1&1&1\\1&0&1\\1&2&3}
-85\detm{1&1&1\\1&0&1\\2&1&0}
\end{aligned}
\]
\end{frame}

\begin{frame}[shrink=17]{Desarrollo completo de $D_x$: menores $3\times3$}
\[
\begin{aligned}
D_x={}&95\left[1\detm{1&0\\2&3}-0\detm{2&0\\1&3}+1\detm{2&1\\1&2}\right]\\
&-125\left[1\detm{1&0\\2&3}-1\detm{2&0\\1&3}+1\detm{2&1\\1&2}\right]\\
&+110\left[1\detm{0&1\\2&3}-1\detm{1&1\\1&3}+1\detm{1&0\\1&2}\right]\\
&-85\left[1\detm{0&1\\1&0}-1\detm{1&1\\2&0}+1\detm{1&0\\2&1}\right]
\end{aligned}
\]
\end{frame}

\begin{frame}[shrink=13]{Desarrollo completo de $D_x$: operaciones internas}
\[
\begin{aligned}
D_x={}&95[(3-0)-0(6-0)+(4-1)]\\
&-125[(3-0)-(6-0)+(4-1)]\\
&+110[(0-2)-(3-1)+(2-0)]\\
&-85[(0-1)-(0-2)+(1-0)]
\end{aligned}
\]

\[
\begin{aligned}
D_x={}&95(6)-125(0)+110(-2)-85(2)\\
D_x={}&570-0-220-170=\boxed{180}
\end{aligned}
\]

\[
x=\frac{D_x}{D}=\frac{180}{4}=\boxed{45}
\]
\end{frame}


\begin{frame}[shrink=8]{Cálculo de $D_y$}
Se reemplaza la columna de $y$ por $B$:
\[
D_y=\detm{
1&95&1&1\\
2&125&0&1\\
1&110&1&0\\
0&85&2&3}
\]

\[
D_y=1(510)-95(8)+1(370)-1(20)
\]

\[
D_y=510-760+370-20=\boxed{100}
\]

\[
y=\frac{D_y}{D}=\frac{100}{4}=\boxed{25}
\]
\end{frame}

% Desarrollo completo agregado para explicar de dónde sale el valor de Dy.
\begin{frame}[shrink=10]{Desarrollo completo de $D_y$: expansión por cofactores}
Se reemplaza la columna de $y$ por el vector $B$:
\[
D_y=\detm{
1&95&1&1\\
2&125&0&1\\
1&110&1&0\\
0&85&2&3}
\]

Se expande por la primera columna:
\[
\begin{aligned}
D_y={}&1\detm{125&0&1\\110&1&0\\85&2&3}
-2\detm{95&1&1\\110&1&0\\85&2&3}\\
&+1\detm{95&1&1\\125&0&1\\85&2&3}
-0\detm{95&1&1\\125&0&1\\110&1&0}
\end{aligned}
\]
\end{frame}

\begin{frame}[shrink=17]{Desarrollo completo de $D_y$: menores $3\times3$}
\[
\begin{aligned}
D_y={}&1\left[125\detm{1&0\\2&3}-110\detm{0&1\\2&3}+85\detm{0&1\\1&0}\right]\\
&-2\left[95\detm{1&0\\2&3}-110\detm{1&1\\2&3}+85\detm{1&1\\1&0}\right]\\
&+1\left[95\detm{0&1\\2&3}-125\detm{1&1\\2&3}+85\detm{1&1\\0&1}\right]
\end{aligned}
\]
\end{frame}

\begin{frame}[shrink=13]{Desarrollo completo de $D_y$: operaciones internas}
\[
\begin{aligned}
D_y={}&125(3-0)-110(0-2)+85(0-1)\\
&-2[95(3-0)-110(3-2)+85(0-1)]\\
&+95(0-2)-125(3-2)+85(1-0)
\end{aligned}
\]

\[
\begin{aligned}
D_y={}&375+220-85-2[285-110-85]-190-125+85\\
D_y={}&510-2(90)-230=510-180-230=\boxed{100}
\end{aligned}
\]

\[
y=\frac{D_y}{D}=\frac{100}{4}=\boxed{25}
\]
\end{frame}


\begin{frame}[shrink=8]{Cálculo de $D_z$}

D_z=\detm{

\[
D_z=1(-360)-1(370)+95(10)-1(160)

\[

\end{frame}

% Desarrollo completo agregado para explicar de dónde sale el valor de Dz.
\begin{frame}[shrink=10]{Desarrollo completo de $D_z$: expansión por cofactores}
Se reemplaza la columna de $z$ por el vector $B$:
\[
D_z=\detm{
1&1&95&1\\
2&1&125&1\\
1&2&110&0\\

Se expande por la primera fila:
\[
\begin{aligned}
D_z={}&1\detm{1&125&1\\2&110&0\\1&85&3}
-1\detm{2&125&1\\1&110&0\\0&85&3}\\
&+95\detm{2&1&1\\1&2&0\\0&1&3}
-1\detm{2&1&125\\1&2&110\\0&1&85}
\end{aligned}
\]
\end{frame}

\begin{frame}[shrink=16]{Desarrollo completo de $D_z$: menores $3\times3$}
\[
\begin{aligned}
D_z={}&\left[1\detm{110&0\\85&3}-2\detm{125&1\\85&3}+1\detm{125&1\\110&0}\right]\\
&-\left[2\detm{110&0\\85&3}-1\detm{125&1\\85&3}+0\detm{125&1\\110&0}\right]\\
&+95\left[2\detm{2&0\\1&3}-1\detm{1&0\\0&3}+1\detm{1&2\\0&1}\right]\\
&-\left[2\detm{2&110\\1&85}-1\detm{1&110\\0&85}+125\detm{1&2\\0&1}\right]
\end{aligned}
\]
\end{frame}

\begin{frame}[shrink=13]{Desarrollo completo de $D_z$: operaciones internas}
\[
\begin{aligned}
D_z={}&[330-2(375-85)+(0-110)]\\
&-[2(330)-1(375-85)+0]\\
&+95[2(6)-1(3)+1(1)]\\
&-[2(170-110)-1(85-0)+125(1)]
\end{aligned}
\]

\[
\begin{aligned}
D_z={}&(-360)-370+95(10)-160\\
D_z={}&-360-370+950-160=\boxed{60}
\end{aligned}
\]

\[
z=\frac{D_z}{D}=\frac{60}{4}=\boxed{15}
\]
\end{frame}

\begin{frame}[shrink=13]{Desarrollo completo de $D_w$: operaciones internas}

\[
w=\frac{D_w}{D}=\frac{40}{4}=\boxed{10}

\end{frame}

\begin{frame}{Optimización del cálculo}
\begin{itemize}
    \item En matrices $4\times4$, conviene expandir por filas o columnas con ceros.
    \item La Regla de Cramer es clara para explicar el procedimiento, pero se vuelve extensa en sistemas grandes.
    \item Para sistemas mayores, se recomienda usar eliminación de Gauss, Gauss-Jordan o software matemático.
    \item El determinante sigue siendo útil porque indica si existe una solución única.
\end{itemize}
